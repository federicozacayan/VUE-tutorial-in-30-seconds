Primitive vs Reference Types

If a strings, boolean or number is send as prop the child componene get a copy of then.

But if we send an object or an array the child component receive a reference.

Event Emiter

src/App.vue

<template>
    <div>
        <app-header v-bind:title="title" v-on:changeTitleFromChild="changeTitleInParent($event)"></app-header>
    </div>
</template>

<script>

    components: {
        'app-header': Header
    },
    data () {
        return {
          title: 'Vue Wizards'
        }

      }
    }
}

    props: {
      title: {
        type: String,
        required: true
      }

        this.$emit('changeTitleInParent', 'Value of title sent from child');

    methods: {
      change123: function(){
        bus.$emit('yearChanged', '2020');
      }
    }
}
</script>
<!-- other.vue component -->
<template>
    <footer>
        <p>Copyright {{ year }}</p>
    </footer>

<script>
// imports
import { bus } from '../main';
export default {
    data() {
        return: {
            year: '2019'
        }
    },
    // it is executed only once at the component creation
    created(){
        // we can listening that event with instance.$on() function
        bus.$on('yearChanged', (data) => {
            this.year = data;
        });
    }
}
</script>
